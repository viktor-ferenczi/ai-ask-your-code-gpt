// Provides summarization and search capability over the source code and documentation of a software project. Facilitates discovering the project gradually, retrieving only the parts relevant for reasoning about the task at hand.
namespace askyourcodedev {

// Start working on a new project
type create = (_: {
// Project contents to download, link to a GitHub repository or a zip/tar.gz archive
url?: string,
}) => any;

// Forget about a project
type delete = (_: {
// Project identifier
project: string,
}) => any;

// Summarize the project, a directory, a file or a code construct (class, method, function, variable)
type summarize = (_: {
// Project identifier
project: string,
// Path to a specific folder or file
path?: string,
// Tail of the file path
tail?: string,
// Simple or fully qualified name of a code construct
name?: string,
}) => any;

// Searches for documentation or a code construct (class, method, function, variable)
type search = (_: {
// Project identifier
project: string,
// Matches the head of the file or folder path (whole project if missing)
path?: string,
// Matches the tail of the file path, for example ".extension" or "filename.extension"
tail?: string,
// Fully qualified or simple name of the code construct
name?: string,
// Free text search expression
text?: string,
}) => any;

} // namespace askyourcodedev
